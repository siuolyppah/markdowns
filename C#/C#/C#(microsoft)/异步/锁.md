# 使用Lock的demo

```cs
namespace Test;
class Progrm
{

    static bool _done;
    static readonly object _locker = new object();

    static void foo()
    {
        lock (_locker)
        {
            if (!_done)
            {
                _done = true;
                Console.WriteLine("it's done");
            }
        }
    }

    static void Main(string[] args)
    {
        new Thread(foo).Start();

        foo();
    }
}
```
